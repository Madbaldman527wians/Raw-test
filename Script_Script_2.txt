local frames=script.Parent.LOCUST:GetChildren()
local decal=script.Parent.the
local function randomsound(folder)
	local get=folder:GetChildren()
	local git=get[math.random(1, #get)]
	return git
end
addsound=function(sound, parent, timeposition)
	local newsound = sound:Clone()
	newsound.Parent = parent
	if timeposition then
		newsound.TimePosition = timeposition
	end
	local newsound2

	newsound:Play()

	spawn(function()
		newsound.Ended:Wait()
		newsound:Destroy()
	end)
	return newsound
end

local glass=1
while true do
	local fmes=#frames
	for frame=0,fmes-1 do
		task.wait(0.02)
		decal.Texture=script.Parent.LOCUST[frame].Texture
		if frame==0 or frame==39 then
			addsound(randomsound(script.Parent.groan), script.Parent).PlaybackSpeed=Random.new():NextNumber(1,1.3)

		end
		if frame==29 or frame==83 then
			print("thump")
			addsound(randomsound(script.Parent.thump), script.Parent).PlaybackSpeed=Random.new():NextNumber(0.66,0.85)
			if not script.Parent:FindFirstChild("glass"..glass) then
				print("BREAK")
				addsound(randomsound(script.Parent.glass), script.Parent).PlaybackSpeed=Random.new():NextNumber(0.8,1)
				for i,v in pairs(script.Parent:GetChildren()) do
					if v:FindFirstChildWhichIsA("Decal") then
						v:Destroy()
					end
				end
				script.Parent.GlassShatter:Emit(165)
                decal.Transparency=1
                return
			else
				script.Parent["glass"..glass].Decal.Transparency=0
				glass+=1
			end

		end
	end
end
